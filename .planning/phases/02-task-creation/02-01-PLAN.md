2. Rewrite the `handleSubmit` function body to call optimistic actions BEFORE awaiting the server:
   ```typescript
   function handleSubmit(e: React.FormEvent) {
     e.preventDefault();
     if (!title.trim() || !workspaceId || selectedMemberIds.length === 0) return;

     const tempId = crypto.randomUUID();
     const trimmedTitle = title.trim();
     const ws = workspaces.find((w) => w.id === workspaceId)!;

     // Build optimistic RawTask
     const optimisticTask: RawTask = {
       id: tempId,
       title: trimmedTitle,
       due_at: dueAt ? `${dueAt}:00Z` : null,
       completed_at: null,
       workspace: { id: ws.id, name: ws.name, kind: ws.kind },
       member_sort_key: 0,
       assignee_count: selectedMemberIds.length,
     };

     // Optimistic actions: fire callback, close modal, reset form, toast
     onTaskCreated?.(optimisticTask);
     resetForm();
     onClose();
     toast("Task created");

     // Server action (fire-and-forget from user's perspective)
     startTransition(async () => {
       try {
         const { subtaskErrors } = await createTaskWithSubtasks({
           title: trimmedTitle,
           description: description.trim() || undefined,
           dueAt: dueAt || undefined,
           workspaceId,
           memberIds: selectedMemberIds,
           subtasks: subtaskRows
             .filter((r) => r.title.trim())
             .map((r) => ({ title: r.title.trim(), dueAt: r.dueAt || undefined })),
         });
         if (subtaskErrors > 0) {
           toast(`Task created, but ${subtaskErrors} subtask(s) could not be saved`, "warning");
         }
       } catch {
         onTaskError?.(tempId);
         toast("Failed to create task", "error");
       }
     });
   }
   ```

   IMPORTANT: `resetForm()` and `onClose()` now happen BEFORE `startTransition` — this is intentional.
   The `toast("Task created")` also fires before server response.
   On error path: `onTaskError` is called with tempId for rollback, modal stays closed (already closed), error toast fires.
